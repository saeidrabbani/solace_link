async function sendSelectedToTelegram() { const checkboxes = document.querySelectorAll(".checkbox:checked"); if (checkboxes.length === 0) { alert("❗ No files selected."); return; } for (const checkbox of checkboxes) { const filename = checkbox.value; try { const res = await fetch("https://solacebot.onrender.com/send-file-to-telegram", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ filename }) }); const result = await res.json(); alert(result.message || `✅ Sent ${filename} to Telegram`); } catch (err) { alert("❌ Failed to send " + filename + ": " + err.message); } } }
